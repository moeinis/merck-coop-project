df['click_date'] = pd.to_datetime(df['click_date'])
df['open_date'] = pd.to_datetime(df['open_date'])
df['sent_date'] = pd.to_datetime(df['sent_date'])

df['open_to_click'] = df['click_date'] - df['open_date']
df['sent_to_click'] = df['click_date'] - df['sent_date']
df['sent_to_open'] = df['open_date'] - df['sent_date']

df['open_to_click'] = df['open_to_click']/np.timedelta64(1,'s')
df['sent_to_click'] = df['sent_to_click']/np.timedelta64(1,'s')
